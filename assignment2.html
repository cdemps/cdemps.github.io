const int buttonPin = 2; // 2 is used as pushbutton pin const int ledPin = 3; // 3 is used as ledPin // variables will change: int buttonState = 0; // variable for reading the pushbutton status void setup() { // makes the LED pin as an output pinMode(ledPin, OUTPUT); // makes the pushbutton pin as an input pinMode(buttonPin, INPUT); // } void loop() { // checks to see if the button is being pushed buttonState = digitalRead(buttonPin); // check if the pushbutton is pressed. If it is, initialize the fade: if (buttonState == HIGH) { // fade in from min to max in increments of 5 points: for (int fadeValue = 0 ; fadeValue <= 255; fadeValue += 5) { // sets the value (range from 0 to 255): analogWrite(ledPin, fadeValue); // wait for 30 milliseconds to see the dimming effect delay(30); } // fade out from max to min in increments of 5 points: for (int fadeValue = 255 ; fadeValue >= 0; fadeValue -= 5) { // sets the value (range from 0 to 255): analogWrite(ledPin, fadeValue); // wait for 30 milliseconds to see the dimming effect delay(30); } } else { // turn LED off when button is not being pressed digitalWrite(ledPin, HIGH); } }
